## Step 2: Create the embedding service

Wire together an engine, a storage backend, and your universes in a shared module.

<CodeBlockTabs defaultValue="openai-pgvector">
  <CodeBlockTabsList>
    <CodeBlockTabsTrigger value="openai-pgvector">OpenAI + pgvector</CodeBlockTabsTrigger>
    <CodeBlockTabsTrigger value="openai-sqlite">OpenAI + sqlite-vec</CodeBlockTabsTrigger>
    <CodeBlockTabsTrigger value="cf-vectorize">Cloudflare AI + Vectorize</CodeBlockTabsTrigger>
  </CodeBlockTabsList>
  <CodeBlockTab value="openai-pgvector">

```ts title="lib/embedding.ts"
import { createEmbeddingService, defineEmbeddable, defineUniverse } from "@justwant/embedding";
import { openAiEmbeddingEngine } from "@justwant/embedding/engines/openai";
import { pgvectorStorageAdapter } from "@justwant/embedding/storages/pgvector";
import OpenAI from "openai";
import { Pool } from "pg";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const articleEmbeddable = defineEmbeddable<{ id: string; title: string; body: string }>({
  idField: "id",
  toText: (a) => `${a.title}\n\n${a.body}`,
  metadataFields: ["title"],
});

const articles = defineUniverse({
  id: "articles",
  dimension: 1536,
  embeddable: articleEmbeddable,
});

export const embedding = createEmbeddingService({
  engine: openAiEmbeddingEngine({ client: openai, model: "text-embedding-3-small" }),
  storage: pgvectorStorageAdapter({ db: pool, tableName: "embeddings", dimension: 1536 }),
  universes: [articles],
});
```

```ts title="worker.ts"
import { createEmbeddingService, defineEmbeddable, defineUniverse } from "@justwant/embedding";
import { cloudflareAiEmbeddingEngine } from "@justwant/embedding/engines/cloudflare-ai";
import { vectorizeStorageAdapter } from "@justwant/embedding/storages/vectorize";

const articleEmbeddable = defineEmbeddable<{ id: string; title: string; body: string }>({
  idField: "id",
  toText: (a) => `${a.title}\n\n${a.body}`,
  metadataFields: ["title"],
});

const articles = defineUniverse({
  id: "articles",
  dimension: 768,
  embeddable: articleEmbeddable,
});

export default {
  async fetch(request: Request, env: Env) {
    const embedding = createEmbeddingService({
      engine: cloudflareAiEmbeddingEngine({ ai: env.AI, dimension: 768 }),
      storage: vectorizeStorageAdapter({ vectorize: env.VECTORIZE_ARTICLES, dimension: 768 }),
      universes: [articles],
    });

    // use embedding service...
  },
};
```
