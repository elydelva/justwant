## Quick setup

Define your domain events, create a bus, and start listening.

```ts title="lib/events.ts"
import { defineEvent, createEventBus } from "@justwant/event";

// 1. Define events
const userCreated = defineEvent("user.created", (id: string, email: string) => ({
  id,
  email,
}));

const orderShipped = defineEvent("order.shipped", (orderId: string) => ({
  orderId,
  shippedAt: new Date(),
}));

// 2. Create the bus
export const bus = createEventBus({
  events: [userCreated, orderShipped] as const,
});
```

```ts title="app.ts"
import { bus } from "@/lib/events";

// Listen for a specific event
bus.listen("user.created", (payload) => {
  console.log("New user:", payload.id, payload.email);
});

// Emit
bus.emit("user.created", "usr_01", "alice@example.com");
```
