## Quick setup

```ts title="lib/prefs.ts"
import {
  definePreference,
  createPreferenceService,
  createMemoryPreferenceAdapter,
} from "@justwant/preference";

// 1. Declare preferences
export const theme = definePreference({ name: "theme", default: "light" });
export const pageSize = definePreference({ name: "pageSize", default: 20 });

// 2. Create the service (swap adapter for production)
export const prefs = createPreferenceService({
  preferences: [theme, pageSize],
  repo: createMemoryPreferenceAdapter(),
});
```
