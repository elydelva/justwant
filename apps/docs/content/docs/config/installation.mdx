## Step 2: Create the config service

Create a shared config instance in a module (e.g. `lib/config.ts`). This example reads from environment variables:

```ts title="lib/config.ts"
import {
  createConfigService,
  defineEnvironment,
  defineValue,
  defineEnvSource,
} from "@justwant/config";

const env = defineEnvSource();

export const config = createConfigService({
  environments: [
    defineEnvironment({
      name: "default",
      sources: {
        databaseUrl: defineValue({ from: env, key: "DATABASE_URL" }),
        port: defineValue({ from: env, key: "PORT" }),
        appSecret: defineValue({ from: env, key: "APP_SECRET" }),
      },
    }),
  ],
});
```

## Step 3: Read values

Import the config instance anywhere in your application:

```ts title="server.ts"
import { config } from "@/lib/config";

const port = await config.get<number>("port");
const dbUrl = await config.get<string>("databaseUrl");

console.log(`Starting on port ${port}`);
```
