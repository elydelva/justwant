## Quick start

```ts title="lib/jobs.ts"
import { createJob, defineJob, defineCron } from "@justwant/job";
import { nodeEngine } from "@justwant/job/engines/node";

const sendWelcomeEmail = defineJob({
  name: "send-welcome-email",
});

const dailyDigest = defineCron({
  job: defineJob({ name: "daily-digest" }),
  cron: "0 9 * * *",
});

const handler = sendWelcomeEmail.handle(async ({ data, logger }) => {
  logger.info("Sending welcome email", { data });
});

const jobs = createJob({
  engine: nodeEngine(),
  crons: [dailyDigest],
  handlers: {
    "daily-digest": dailyDigest.job.handle(async ({ logger }) => {
      logger.info("Sending daily digest");
    }),
  },
});

await jobs.start();

// Manually trigger
await jobs.enqueue(sendWelcomeEmail, { userId: "u-1" });
```
