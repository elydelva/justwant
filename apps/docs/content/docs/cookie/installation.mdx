## Step 2: Define your first cookie

Define a typed cookie with a name and optional schema. The cookie definition is reusable across your codebase.

```ts title="lib/cookies.ts"
import { defineCookie } from "@justwant/cookie";
import { z } from "zod";

export const themeCookie = defineCookie("theme", {
  schema: z.enum(["light", "dark"]),
  default: "light",
  onMismatch: "fallback",
});
```

## Step 3: Choose an adapter

Pick the adapter that matches your environment and wire it up:

  <CodeBlockTab value="nextjs">

```ts title="app/actions.ts"
import { cookies } from "next/headers";
import { NextJSAdapter, createCookieStore } from "@justwant/cookie";
import { themeCookie } from "@/lib/cookies";

export async function getTheme() {
  const cookieStore = await cookies();
  const adapter = NextJSAdapter(cookieStore);
  const store = createCookieStore({ theme: themeCookie }, { adapter });
  return store.get();
}
```

  <CodeBlockTab value="browser">

```ts title="lib/cookies-browser.ts"
import { DocumentAdapter, createCookieStore } from "@justwant/cookie";
import { themeCookie } from "@/lib/cookies";

const adapter = DocumentAdapter();
export const store = createCookieStore({ theme: themeCookie }, { adapter });

// Read
const { theme } = store.get(); // "light" | "dark"

// Write
store.set("theme", "dark");
```

  </CodeBlockTab>
  <CodeBlockTab value="fetch">

```ts title="middleware.ts"
import { RequestResponseAdapter, createCookieStore } from "@justwant/cookie";
import { themeCookie } from "@/lib/cookies";

export function middleware(request: Request): Response {
  const response = new Response();
  const adapter = RequestResponseAdapter(request, response);
  const store = createCookieStore({ theme: themeCookie }, { adapter });

  const { theme } = store.get();
  console.log("Current theme:", theme);

  return response;
}
```

  </CodeBlockTab>
  <CodeBlockTab value="header">

```ts title="server.ts"
import { HeaderAdapter, createCookieStore } from "@justwant/cookie";
import { themeCookie } from "@/lib/cookies";

function handler(req: { headers: { cookie?: string } }) {
  const responseHeaders: string[] = [];
  const adapter = HeaderAdapter({
    getCookie: () => req.headers.cookie ?? "",
    appendSetCookie: (value) => responseHeaders.push(value),
  });
  const store = createCookieStore({ theme: themeCookie }, { adapter });

  const { theme } = store.get();
  store.set("theme", "dark");

  return { theme, setCookieHeaders: responseHeaders };
}
```
